Notifications

Full-layout notification


$$('#open-full').on('click', function () { var notificationFull = app.notification.create({ icon: '7', title: 'Framework7', titleRightText: 'now', subtitle: 'This is a subtitle', text: 'This is a simple notification message', closeTimeout: 3000, }).open(); });

With Close Button


$$('#open-with-button').on('click', function () { var notificationWithButton = app.notification.create({ icon: '7', title: 'Framework7', subtitle: 'Notification with close button', text: 'Click (x) button to close me', closeButton: true, }).open(); });

Click to Close


$$('#open-click-to-close').on('click', function () { var notificationWithButton = app.notification.create({ icon: '7', title: 'Framework7', subtitle: 'Notification with close button', text: 'Click (x) button to close me', closeButton: true, }).open(); });

Callback on Close


$$('#open-callback-on-close').on('click', function () { var notificationCallbackOnClose = app.notification.create({ icon: '7', title: 'Framework7', titleRightText: 'now', subtitle: 'Notification with close on click', text: 'Click me to close', closeOnClick: true, on: { close: function () { app.dialog.alert('Notification closed'); }, }, }).open(); });
